Vite路径别名配置

# 一、安装 ts 类型定义文件的包

npm install @types/node --save-dev

# 二、在vite.config.ts文件中加入

import { resolve } from 'path'

export default defineConfig({
  plugins: [...],
  resolve: {
    // 配置路径别名
    alias: [
      {
        find: '@', // 指向的是src目录
        replacement: resolve(__dirname, 'src')
      }
    ]
  }
})

# 三、在tsconfig.json文件中的compilerOptions下加入即可

{
  "compilerOptions": {
    ...
    "baseUrl": ".",
    "paths": {
      "@/*": [
        "./src/*"
      ]
    }
  }
}
